🗄️ SQL Analysis — Top Performing Apps
Run Query
-- Top categories by avg success score
SELECT
  category,
  COUNT(*) AS total_apps,
  ROUND(AVG(rating), 2) AS avg_rating,
  SUM(installs) AS total_installs,
  AVG(success_score) AS avg_score
FROM android_apps
WHERE rating IS NOT NULL
GROUP BY category
ORDER BY avg_score DESC
LIMIT 10;
